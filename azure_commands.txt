
## To create resource group.

abc@DESKTOP-9QH4IRU:~$ az group create --name ABC-RG1 --location hist

  "id": "/subscriptions/0b715e4f-72d7-493c-95b4-a10cbb176546/resourceGroups/ABC-RG1",

    "provisioningState": "Succeeded"

#### create another resrouce group in different location

az group create --name FutruesEast --location eastus
{
  "id": "/subscriptions/0b715e4f-72d7-493c-95b4-a10cbb176546/resourceGroups/FutruesEast",
  "location": "eastus",
  "managedBy": null,
  "name": "FutruesEast",
  "properties": {
    "provisioningState": "Succeeded"
  },
  "tags": null,
  "type": "Microsoft.Resources/resourceGroups"
}


### list the resource group 
az group list -o table
Name          Location    Status
------------  ----------  ---------
ABC-RG1       westus2     Succeeded
Futures-East  eastus2     Succeeded
FutruesEast   eastus      Succeeded

### delete the resource group.

az group delete --name futures-East --yes --no-wai
